import math def generate_cubic_star_path(cx=50, cy=50, Ro=46, Rc=5): # Ro: outer radius (tips) # Rc: control point radius (along the ray of the tips) path_parts = [] num_points = 8 for k in range(num_points): theta_k = math.radians(-90 + k * 360 / num_points) theta_next = math.radians(-90 + (k + 1) * 360 / num_points) # Tip k tx = cx + Ro * math.cos(theta_k) ty = cy + Ro * math.sin(theta_k) # Next Tip k+1 tx_next = cx + Ro * math.cos(theta_next) ty_next = cy + Ro * math.sin(theta_next) # Control point 1 (on the ray of Tip k) c1x = cx + Rc * math.cos(theta_k) c1y = cy + Rc * math.sin(theta_k) # Control point 2 (on the ray of Tip k+1) c2x = cx + Rc * math.cos(theta_next) c2y = cy + Rc * math.sin(theta_next) if k == 0: path_parts.append(f"M {tx:.2f} {ty:.2f}") # Cubic Bezier from Tip k to Tip k+1 with control points c1 and c2 path_parts.append(f"C {c1x:.2f} {c1y:.2f} {c2x:.2f} {c2y:.2f} {tx_next:.2f} {ty_next:.2f}") path_parts.append("Z") return " ".join(path_parts) # Let's print different values of Rc for rc in [0, 4, 8, 12, 16]: print(f"Rc={rc}:") print(generate_cubic_star_path(Ro=46, Rc=rc)) print()
